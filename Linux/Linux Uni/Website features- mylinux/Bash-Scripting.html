# This script prints a welcome message and lists the home directory.

# Get the current username

# Display the welcome message

# List the contents of the home directory

Exercise 3: Create and Move Files üìÇ

Create a new script:

nano file_movement.sh

Write script:

#!/bin/bash
# This script creates two directories, five files, moves three files, and prints folder contents.

# Create two directories
mkdir folderA folderB

# Create five files in folderA
touch folderA/file{1..5}.txt

# Move three files from folderA to folderB
mv folderA/file{1..3}.txt folderB/

# Print the contents of both folders
echo "Contents of folderA:"; ls folderA
echo "Contents of folderB:"; ls folderB

Run the script:

chmod +x file_movement.sh
./file_movement.sh

Exercise 4: Using Globbing to Move Files üåç

Modify previous script:

nano globbing_script.sh

Write script:

#!/bin/bash
# This script moves all files from one folder to another using globbing.

# Create directories
mkdir folderX folderY

# Create five files in folderX
touch folderX/file{1..5}.txt

# Move all files at once using globbing
mv folderX/* folderY/

# Rename folderY
mv folderY renamed_folder

# Change into the renamed folder
cd renamed_folder

# Grant read and execute permissions to the group
chmod g+rx *

# Print the folder contents
ls -l

Run the script:

chmod +x globbing_script.sh
./globbing_script.sh

Exercise 5: Interactive File Move Script üñ±Ô∏è

Create the script:

nano interactive_move.sh

Write script:

#!/bin/bash
# This script moves a file to another folder and asks for confirmation before overwriting.

# Prompt user for file name
read -p "Enter the file name to move: " file_name

# Prompt user for destination folder
read -p "Enter the destination folder path: " dest_folder

# Check if file exists
if [ ! -f "$file_name" ]; then echo "Error: File does not exist!"; exit 1; fi

# Check if destination folder exists
if [ ! -d "$dest_folder" ]; then echo "Error: Destination folder does not exist!"; exit 1; fi

# Check if file with same name exists in destination
if [ -f "$dest_folder/$file_name" ]; then
    read -p "A file with the same name exists. Overwrite? (y/n): " confirmation
    if [ "$confirmation" != "y" ]; then echo "Operation canceled."; exit 0; fi
fi

# Move the file
mv "$file_name" "$dest_folder/"

echo "File moved successfully!"

Run the script:

chmod +x interactive_move.sh
./interactive_move.sh
